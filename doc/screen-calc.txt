gapWidth = buttonWidth * gapPercentage
screenWidth = buttonColumns * buttonWidth + (buttonColumns + 1) * gapWidth
   = buttonColumns * buttonWidth + (buttonColumns + 1) * buttonWidth * gapPercentage
   = buttonColumns * buttonWidth + buttonColumns * buttonWidth * gapPercentage + buttonWidth * gapPercentage
   = buttonColumns * (buttonWidth + buttonWidth * gapPercentage) + buttonWidth * gapPercentage
screenWidth - buttonWidth * gapPercentage = buttonColumns * buttonWidth * (gapPercentage + 1)
buttonColumns = (screenWidth - buttonWidth * gapPercentage) / (buttonWidth * (gapPercentage + 1))

screenWidth = 16
buttonWidth = 4
gapPercentage = 0.25
buttonColumns = (16 - 4 * 0.25) / (4 * 1.25)
   = (16 - 1) / 5
   = 15 / 5
   = 3

screenWidth = buttonColumns * buttonWidth + (buttonColumns + 1) * buttonWidth * gapPercentage
            = buttonWidth * (buttonColumns + (buttonColumns + 1) * gapPercentage)
            = buttonWidth * (buttonColumns + buttonColumns * gapPercentage + gapPercentage)
            = buttonWidth * (buttonColumns * (1 + gapPercentage) + gapPercentage)
buttonWidth = screenWidth / (buttonColumns * (1 + gapPercentage) + gapPercentage)
			
screenWidth = 16
buttonColumns = 3
gapPercentage = 0.25
buttonWidth = 16 / (3 * 1.25 + 0.25)
   = 16 / 4
   = 4
